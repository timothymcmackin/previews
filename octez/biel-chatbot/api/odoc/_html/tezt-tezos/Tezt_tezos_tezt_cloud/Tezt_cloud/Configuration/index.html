type docker_image =
  1. | Gcp of {
    1. alias : string;
    }
  2. | Octez_release of {
    1. tag : string;
    }
type t = private {
  1. machine_type : string;
  2. docker_image : docker_image;
  3. max_run_duration : int option;
  4. binaries_path : string;
  5. os : string;
}
val make : ?os:string -> ?binaries_path:string -> ?max_run_duration:int -> ?machine_type:string -> ?docker_image:docker_image -> unit -> t

make ?machine_type () is a smart-constructor to make a VM configuration.

Default value for max_run_duration is 7200.

Default value for machine_type is n1-standard-2.

Default value for docker_image is Custom {tezt_cloud} where tezt_cloud is the value provided by the environement variable $TEZT_CLOUD.
