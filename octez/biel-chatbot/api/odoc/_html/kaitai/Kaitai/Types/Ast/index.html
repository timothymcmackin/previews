Module Types.Ast

Ast module defines Kaitai Struct expression language.

  • For more about it see the Kaitai Struct User Guide.
  • For a reference implementation see.
type boolop =
  1. | Or
  2. | And
type typeId = {
  1. absolute : bool;
  2. names : string list;
  3. isArray : bool;
}
val empty_typeId : typeId
type operator =
  1. | Add
  2. | Sub
  3. | Mult
  4. | Div
  5. | Mod
  6. | LShift
  7. | RShift
  8. | BitOr
  9. | BitXor
  10. | BitAnd
type unaryop =
  1. | Invert
    (*

    Bitwise negation operator. Applicable only to IntNums.

    *)
  2. | Not
    (*

    Boolean negation operator. Applicable only to Booleans.

    *)
  3. | Minus
    (*

    Arithmetic negation operator. Applicable only to IntNums or FloatNums.

    *)
type cmpop =
  1. | Eq
  2. | NotEq
  3. | Lt
  4. | LtE
  5. | Gt
  6. | GtE
type t =
  1. | Raw of string
    (*

    Temporary: Raw So that we don't need to deal with parsing/printing of Ast.expr.

    *)
  2. | BoolOp of {
    1. op : boolop;
    2. values : t list;
    }
    (*

    BoolOp represent a boolean operation.

    *)
  3. | BinOp of {
    1. left : t;
    2. op : operator;
    3. right : t;
    }
    (*

    BinOp represents a binary operation.

    *)
  4. | UnaryOp of {
    1. op : unaryop;
    2. operand : t;
    }
    (*

    UnaryOp represents an unary operation.

    *)
  5. | IfExp of {
    1. condition : t;
    2. ifTrue : t;
    3. ifFalse : t;
    }
  6. | Compare of {
    1. left : t;
    2. ops : cmpop;
    3. right : t;
    }
    (*

    Represents X < Y, X > Y and so on.

    *)
  7. | Call of {
    1. func : t;
    2. args : t list;
    }
    (*

    Call represents a function call.

    *)
  8. | IntNum of int
    (*

    IntNum is originally represented by math.BigInt.

    *)
  9. | FloatNum of float
    (*

    FloatNum is originally represented by math.BigDecimal.

    *)
  10. | Str of string
    (*

    Str is a string expression.

    *)
  11. | Bool of bool
    (*

    Bool is a bool expression.

    *)
  12. | EnumByLabel of {
    1. enumName : Identifier.t;
    2. label : Identifier.t;
    3. inType : typeId;
    }
    (*

    EnumByLabel represents an enum istance.

    *)
  13. | EnumById of {
    1. enumName : Identifier.t;
    2. id : t;
    3. inType : typeId;
    }
    (*

    EnumById represents an enum istance.

    *)
  14. | Attribute of {
    1. value : t;
    2. attr : Identifier.t;
    }
    (*

    Attribute represents a x.y where y is attr.

    *)
  15. | CastToType of {
    1. value : t;
    2. typeName : typeId;
    }
    (*

    CastToType is used for casting types.

    *)
  16. | ByteSizeOfType of {
    1. typeName : typeId;
    }
    (*

    BytesSizeOfType expression for specifying type size in bytes.

    *)
  17. | BitSizeOfType of {
    1. typeName : typeId;
    }
    (*

    BitSizeOfType expression for specifying type size in bits.

    *)
  18. | Subscript of {
    1. value : t;
    2. idx : t;
    }
    (*

    Represents [X[Y]].

    *)
  19. | Name of Identifier.t
    (*

    Name is used for defining variable, e.g. x.

    *)
  20. | List of t list
    (*

    List represents a list.

    *)
type expr = t
val to_string : t -> string
