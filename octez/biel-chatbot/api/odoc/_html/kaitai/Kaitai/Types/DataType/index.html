Module Types.DataType

DataType module defines AST for describing underlying data types.

For a reference implementation see.

type data_type =
  1. | NumericType of numeric_type
  2. | BooleanType of boolean_type
  3. | BytesType of bytes_type
  4. | StrType of str_type
  5. | ComplexDataType of complex_data_type
  6. | AnyType
  7. | Raw of string
and int_width =
  1. | W1
  2. | W2
  3. | W4
  4. | W8
and numeric_type =
  1. | Int_type of int_type
  2. | Float_type of float_type
and int_type =
  1. | CalcIntType
  2. | Int1Type of {
    1. signed : bool;
    }
  3. | IntMultiType of {
    1. signed : bool;
    2. width : int_width;
    3. endian : Endianness.fixed_endian option;
    }
  4. | BitsType of {
    1. width : int;
    2. bit_endian : BitEndianness.t;
    }
and float_type =
  1. | CalcFloatType
  2. | FloatMultiType of {
    1. width : int_width;
    2. endian : Endianness.fixed_endian option;
    }
and boolean_type =
  1. | BitsType1 of BitEndianness.t
  2. | CalcBooleanType
and bytes_type =
  1. | CalcBytesType
  2. | BytesEosType of {
    1. terminator : int option;
    2. include_ : bool;
    3. padRight : int option;
    4. mutable process : processExpr option;
    }
  3. | BytesLimitType of {
    1. size : Ast.expr;
    2. terminator : int option;
    3. include_ : bool;
    4. padRight : int option;
    5. mutable process : processExpr option;
    }
  4. | BytesTerminatedType of {
    1. terminator : int;
    2. include_ : bool;
    3. consume : bool;
    4. eosError : bool;
    5. mutable process : processExpr option;
    }
and str_type =
  1. | CalcStrType
  2. | StrFromBytesType of {
    1. bytes : bytes_type;
    2. encoding : string;
    }
and array_type =
  1. | ArrayTypeInStream
  2. | CalcArrayType
and complex_data_type =
  1. | StructType
  2. | UserType of string
  3. | ArrayType of array_type
and switch_type = {
  1. on : Ast.expr;
  2. cases : (Ast.expr * data_type) list;
  3. isOwning : bool;
  4. mutable isOwningInExpr : bool;
}
type t = data_type
val to_string : t -> string
