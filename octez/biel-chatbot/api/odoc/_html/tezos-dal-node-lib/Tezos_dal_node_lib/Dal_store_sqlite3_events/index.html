val applied_migration : name:string -> duration:float -> unit Lwt.t

val migrations_from_the_future : applied:int -> known:int -> unit Lwt.t

migration_from_the_future ~applied ~known advertizes that there are more migrations applied to the store than known from the DAL node, which suggests the DAL node is outdated.
