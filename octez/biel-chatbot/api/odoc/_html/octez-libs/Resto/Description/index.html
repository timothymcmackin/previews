Service directory description

type request = {
  1. recurse : bool;
}
type 'schema service = {
  1. description : string option;
  2. path : path_item list;
  3. meth : meth;
  4. query : query_item list;
  5. input : 'schema Stdlib.Lazy.t option;
  6. output : 'schema Stdlib.Lazy.t;
  7. error : 'schema Stdlib.Lazy.t;
}
and path_item =
  1. | PStatic of string
  2. | PDynamic of Arg.descr
  3. | PDynamicTail of Arg.descr
and query_item = {
  1. name : string;
  2. description : string option;
  3. kind : query_kind;
}
and query_kind =
  1. | Single of Arg.descr
  2. | Optional of Arg.descr
  3. | Flag
  4. | Multi of Arg.descr
type 'schema directory =
  1. | Empty
  2. | Static of 'schema static_directory
  3. | Dynamic of string option
and 'schema static_directory = {
  1. services : 'schema service MethMap.t;
  2. subdirs : 'schema static_subdirectories option;
}
and 'schema static_subdirectories =
  1. | Suffixes of 'schema directory StringMap.t
  2. | Arg of Arg.descr * 'schema directory
val pp_print_directory : Stdlib.Format.formatter -> 'schema directory -> unit
