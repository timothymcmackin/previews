Module Batcher_worker_types.Request

type ('a, 'b) t =
  1. | Register : {
    1. order : Z.t option;
    2. messages : string list;
    3. drop_duplicate : bool;
    } -> (Octez_smart_rollup.L2_message.id list, Tezos_base.TzPervasives.error Tezos_base.TzPervasives.trace) t
    (*

    Request to register new L2 messages in the queue. if drop_duplicate is true, then the elements of messages already processed by the batcher, with drop_duplicate = true, are dropped.

    *)
  2. | Produce_batches : (unit, Tezos_base.TzPervasives.error Tezos_base.TzPervasives.trace) t
    (*

    Request to produce new messages batches and submit them to the injector.

    *)
  3. | Clear_queues : (unit, Tezos_base.TzPervasives.error Tezos_base.TzPervasives.trace) t
    (*

    Request to clear all queues, the heap and the batched messages.

    *)
  4. | Remove_messages : order_request -> (unit, Tezos_base.TzPervasives.error Tezos_base.TzPervasives.trace) t
    (*

    Request to remove messages in the heap and batched with order inferior to order_below. If drop_no_order is set, remove also all messages that have no order specified.

    *)
